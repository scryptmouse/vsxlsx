 11: function alphacize($idx)
 12: {
 13:     for($r = ""; $idx >= 0; $idx = intval($idx / 26) - 1)
 14:         $r = chr($idx%26 + 0x61) . $r;
 15:     return $r;

 23:  * @return  int
 24:  */
 25: function indicize($alpha)

 29:     $n = 0;

 32:         $n = $n*26 + ord($alpha[$i]) - 0x60;
 33:

100:  * Recursively remove a directory, equivalent to `rm -rf`.
101:  *
102:  * @return  boolean
103:  */
104: function recursive_rm($filename) {
105:     $writable = is_writable($filename);
106: 
107:     if ($writable && is_dir($filename)) {
108:         $can_delete_this_directory = true;
109:         $files = array_diff(scandir($filename), array('.', '..'));
110: 
111:         foreach($files as $file)
112:         {
113:             if (!recursive_rm($filename.'/'.$file))
114:                 $can_delete_this_directory = false;
115:         }
116: 
117:         return $can_delete_this_directory ? rmdir($filename) : false;
118:     } elseif ($writable && is_file($filename)) {
119:         return unlink($filename);
120:     } else {
121:         return false;
122:     }

368:     protected function process_cell($cell)
369:     {
370:         $v = (string) $cell->v;
371: 
372:         // If it has a 't' (type) of 's' (string), use the value to look up string value.
373:         if (isset($cell['t']) && $cell['t'] == 's') {

376:

380:             foreach($si->xpath('.//n:t') as $t)

383:             }
